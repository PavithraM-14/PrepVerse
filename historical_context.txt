## Elements（界面元素）

- **按钮设计**：主按钮高度 44px，圆角 8px，填充主色，悬停时亮度提升 10%；次要按钮使用描边样式，避免视觉干扰

- **徽章与标签**：圆角 6px，高度 24px，填充功能色，文字白色，用于状态标识（如"已完成""进行中"）

## Motion（动效设计）

- **页面切换**：使用 200ms 淡入淡出（opacity 0→1），避免生硬跳转
- **卡片悬停**：Y 轴上移 4px + 投影加深（`0 8px 20px rgba(0,0,0,0.25)`），过渡时间 150ms
- **按钮点击**：缩放至 0.95 倍（transform: scale(0.95)），持续 100ms，释放后回弹
- **XP 增长动效**：进度条填充使用 500ms 缓动动画（ease-out），同时数字从旧值滚动至新值
- **成就解锁**：徽章从中心放大（scale 0→1.2→1），持续 400ms，配合霓虹紫光晕闪烁一次
- **AI 回复**：文字逐字浮现（打字机效果），速度 30ms/字符，营造实时对话感
- **加载状态**：使用旋转的圆环加载器（主色描边），避免骨架屏或进度条

## 技术建议

- 使用 Tailwind CSS 的 `transition` 和 `transform` 工具类实现动效，避免引入额外动画库
- 深色/浅色模式通过 CSS 变量切换，确保一键切换无闪烁
- 移动端使用 `touch-action` 优化触摸响应，按钮最小点击区域 44×44px
- 图片懒加载 + WebP 格式优化加载速度，AI 头像使用 SVG 或低分辨率占位符

   260 export interface SSEOptions {
   261   onData: (data: string) => void;
   262   onEvent?: (event: unknown) => void;

   267 export function createSSEHook(options: SSEOptions): AfterResponseHook {
   268   return async (request, _opts, response) => {
   269     if (!response.ok || !response.body) return;
   270 
   271     let done = false;
   272     const finish = (err?: Error) => {
   273       if (!done) { done = true; options.onCompleted?.(err); }
   274     };
   275 
   276     const reader = response.body.getReader();
   277     const decoder = new TextDecoder("utf8");
   278     const parser = createParser({
   279       onEvent: (event) => {
   280         if (!event.data) return;
   281         options.onEvent?.(event);
   282         for (const chunk of event.data.split("\n")) options.onData(chunk);
   283       },
   284     });
   285 
   286     const read = (): void => {
   287       reader.read().then(({ done: streamDone, value }) => {
   288         if (streamDone) { finish(); return; }
   289         parser.feed(decoder.decode(value, { stream: true }));
   290         read();
   291       }).catch((err) => {
   292         if (request.signal.aborted) { options.onAborted?.(); return; }
   293         finish(err as Error);
   294       });
   295     };
   296     read();
   297     return response;
   298   };

import ky, { type AfterResponseHook } from 'ky';
import { createParser } from 'eventsource-parser';

export interface SSEOptions {
  onData: (data: string) => void;
  onEvent?: (event: unknown) => void;
  onCompleted?: (error?: Error) => void;
  onAborted?: () => void;
}

export function createSSEHook(options: SSEOptions): AfterResponseHook {
  return async (request, _opts, response) => {
    if (!response.ok || !response.body) return;

    let done = false;
    const finish = (err?: Error) => {
      if (!done) {
        done = true;
        options.onCompleted?.(err);
      }
    };

    const reader = response.body.getReader();
    const decoder = new TextDecoder('utf8');
    const parser = createParser({
      onEvent: (event) => {
        if (!event.data) return;
        options.onEvent?.(event);
        for (const chunk of event.data.split('\n')) options.onData(chunk);
      },
    });

    const read = (): void => {
      reader
        .read()
        .then(({ done: streamDone, value }) => {
          if (streamDone) {
            finish();
            return;
          }
          parser.feed(decoder.decode(value, { stream: true }));
          read();
        })
        .catch((err) => {
          if (request.signal.aborted) {
            options.onAborted?.();
            return;
          }
          finish(err as Error);
        });
    };
    read();
    return response;
  };
}

export interface StreamRequestOptions {
  functionUrl: string;
  requestBody: unknown;
  supabaseAnonKey: string;
  onData: (data: string) => void;
  onComplete: () => void;
  onError: (error: Error) => void;
  signal?: AbortSignal;
}

export async function sendStreamRequest(options: StreamRequestOptions): Promise<void> {
  const { functionUrl, requestBody, supabaseAnonKey, onData, onComplete, onError, signal } = options;

  const sseHook = createSSEHook({
    onData,
    onCompleted: (err) => (err ? onError(err) : onComplete()),
    onAborted: () => console.log('Stream aborted'),
  });

  try {
    await ky.post(functionUrl, {
      json: requestBody,
      headers: {
        Authorization: `Bearer ${supabaseAnonKey}`,
        apikey: supabaseAnonKey,
        'Content-Type': 'application/json',
      },
      signal,
      timeout: 120000,
      hooks: { afterResponse: [sseHook] },
    });
  } catch (err) {
    if (!signal?.aborted) onError(err as Error);
  }
}
